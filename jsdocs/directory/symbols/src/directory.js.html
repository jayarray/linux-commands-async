1 let REMOVE = require('./remove.js');
  2 let MKDIR = require('./mkdir.js');
  3 let PATH = require('./path.js');
  4 let DISKUSAGE = require('./diskusage.js');
  5 let VALIDATE = require('./validate.js');
  6 
  7 //-----------------------------------------
  8 // DIRECTORY
  9 
 10 /**
 11  * Delete a directory.
 12  * @param {string} path
 13  * @param {Command} executor Command object that will execute the command.
 14  * @returns {Promise} Returns a promise that will resolve if successful, otherwise it returns an error.
 15  */
 16 function Remove(path, executor) {
 17   return REMOVE.Directories([path], executor);
 18 }
 19 
 20 /**
 21  * Create a directory.
 22  * @param {string} path
 23  * @param {Command} executor Command object that will execute the command.
 24  * @returns {Promise} Returns a promise that will resolve if successful, otherwise it returns an error.
 25  */
 26 function Create(path, executor) {
 27   let pathError = VALIDATE.IsStringInput(path);
 28   if (pathError)
 29     return Promise.reject(`Failed to create directory: ${pathError}`);
 30 
 31   if (!executor)
 32     return Promise.reject(`Failed to create directory: Executor is required`);
 33 
 34   return new Promise((resolve, reject) => {
 35     MKDIR.MakeDirectory(path, executor).then(success => {
 36       resolve();
 37     }).catch(error => reject(`Failed tp create directory: ${error}`));
 38   });
 39 }
 40 
 41 /**
 42  * Determine folder size.
 43  * @param {string} path
 44  * @param {Command} executor Command object that will execute the command.
 45  * @returns {Promise<number>} Returns a promise. If it resolves, it returns an integer representing the size in bytes. Else, it rejects and returns an error.
 46  */
 47 function Size(path, executor) {
 48   let pathError = VALIDATE.IsStringInput(path);
 49   if (pathError)
 50     return Promise.reject(`Failed to get directory size: ${pathError}`);
 51 
 52   if (!executor)
 53     return Promise.reject(`Failed to get directory size: Executor is required`);
 54 
 55   return new Promise((resolve, reject) => {
 56     DISKUSAGE.DirSize(path, executor).then(size => {
 57       resolve(size);
 58     }).catch(reject);
 59   });
 60 }
 61 
 62 //--------------------------------------
 63 // EXPORTS
 64 
 65 exports.Remove = Remove;
 66 exports.Create = Create;
 67 exports.Size = Size;
