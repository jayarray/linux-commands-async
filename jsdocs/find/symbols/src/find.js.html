  7 /**
  8  * List all files matching the given name pattern.
  9  * @param {string} path Directory location
 10  * @param {string} pattern
 11  * @param {Number} maxDepth Maximum number of levels to recurse
 12  * @param {Command} executor Command object that will execute the command.
 13  * @returns {Promise<Array<string>>} Returns a promise. If it resolves, it returns an array of filepaths. Else, it returns an error.
 14  */
 15 function FilesByName(path, pattern, maxDepth, executor) {
 16   let error = VALIDATE.IsStringInput(path);
 17   if (error)
 18     return Promise.reject(`Failed to find files by name: path is ${error}`);
 19 
 20   error = PatternValidator(pattern);
 21   if (error)
 22     return Promise.reject(`Failed to find files by name: ${error}`);
 23 
 24   error = MaxDepthValidator(maxDepth);
 25   if (error)
 26     return Promise.reject(`Failed to find files by name: ${error}`);
 27 
 28   if (!executor)
 29     return Promise.reject(`Failed to find files by name: Executor is required`);
 30 
 31   return new Promise((resolve, reject) => {
 32     let args = [path];
 33     if (maxdepth)
 34       args.push('-maxdepth', maxDepth);
 35     args.push('-type', 'f', '-name', pattern);
 36 
 37     executor.Execute('find', args).then(output => {
 38       let deniedPaths = [];
 39 
 40       if (output.stderr) {
 41         if (output.stderr.includes('Permission denied')) {
 42           let lines = output.stderr.split('\n').filter(line => line && line.trim() != '' && line != path);
 43 
 44           for (let i = 0; i < lines.length; ++i) {
 45             let currLine = lines[i];
 46             let dPath = currLine.split(':')[1].trim();
 47             deniedPaths.push(dPath);
 48           }
 49         }
 50         else {
 51           reject(`Failed to find files by name: ${output.stderr}`);
 52           return;
 53         }
 54       }
 55 
 56       let paths = output.stdout.split('\n').filter(line => line && line.trim() != '' && line != path && line != path);
 57       resolve({ paths: paths, denied: deniedPaths });
 58     }).catch(error => reject(`Failed to find files by name: ${error}`));
 59   });
 60 }

370 function ArgsValidator(args) {
371   let error = VALIDATE.IsArray(args);
372   if (error)
373     return `args is ${error}`;
374 
375   for (let i = 0; i < args.length; ++i) {
376     let currArg = args[i];
377 
378     let argIsValidString = VALIDATE.IsStringInput(currArg) == null;
379     let argIsValidNumber = !isNaN(currArg);
380 
381     if (!argIsValidString && !argIsValidNumber)
382       return `Arg elements must be string or number type`;
383   }
384 
385   return null;
386 }
387 
388 function MaxDepthValidator(maxDepth) {
389   if (maxDepth == null)
390     return null;
391 
392   let error = VALIDATE.IsInteger(maxDepth);
393   if (error)
394     return `MaxDepth is ${error} `;
395 
396   let min = 0;
397   error = VALIDATE.IsIntegerInRange(maxDepth, min, null);
398   if (error)
399     return `MaxDepth is ${error} `;
400 
401   return null;
402 }
403 
404 function PatternValidator(pattern) {
405   let error = VALIDATE.IsInstance(pattern);
406   if (error)
407     return `Pattern is ${error} `;
408 
409   if (typeof pattern != 'string')
410     return 'Pattern must be string type';
411 
412   if (pattern == '')
413     return 'Pattern cannot be empty';
414 
415   return null;
416 }
417 
418 function TextValidator(text) {
419   let error = VALIDATE.IsInstance(text);
420   if (error)
421     return `Text is ${error} `;
422 
423   if (typeof text != 'string')
424     return 'Text must be string type';
425 
426   if (text == '')
427     return 'Text cannot be empty';
428 
429   return null;
430 }
