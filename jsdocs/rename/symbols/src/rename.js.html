1 let _path = require('path');
  2 
  3 let PATH = require('./path.js');
  4 let MOVE = require('./move.js');
  5 let VALIDATE = require('./validate.js');

  8 // RENAME
  9 
 10 /**
 11  * Change name of file or directory.
 12  * @param {string} src Source
 13  * @param {string} newName New desired name file or directory.
 14  * @param {Command} executor Command object that will execute the command.
 15  * @returns {Promise} Returns a promise that resolves if successful, otherwise rejects and returns an error.
 16  */
 17 function Rename(src, newName, executor) {
 18   let srcError = VALIDATE.IsStringInput(src);
 19   if (srcError)
 20     return Promise.reject(`Failed to rename: Source is ${srcError}`);
 21 
 22   let newNameIsValid = typeof newName == 'string' && newName != '';
 23   if (!newNameIsValid)
 24     return Promise.reject(`Failed to rename: New name must be a string type and cannot be empty`);
 25 
 26   if (!executor)
 27     return Promise.reject(`Failed to rename: Executor is required`);
 28 
 29   return new Promise((resolve, reject) => {
 30     let parentDir = PATH.ParentDir(src);
 31     let dest = _path.join(parentDir, newName);
 32 
 33     MOVE.Move(src, dest, executor).then(success => {
 34       resolve();
 35     }).catch(error => reject(`Failed to rename: ${error}`));
 36   });
 37 }
 38

 41 
 42 exports.Rename = Rename;
