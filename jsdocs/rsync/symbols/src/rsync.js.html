  6 function argsValidator(args) {
  7   let error = VALIDATE.IsArray(args);
  8   if (error)
  9     return `arguments are ${error}`;
 10 
 11   // Every element should be a number or 
 12   if (!args.every(currArg => !isNaN(currArg) || VALIDATE.IsStringInput(currArg) == null))
 13     return `arg elements must be string or number type`;
 14 
 15   return null;
 16 }

 19 // RSYNC
 20 
 21 function Rsync_(user, host, flag, src, dest, executor) { // PRIVATE
 22   let userError = VALIDATE.IsStringInput(user);
 23   if (userError)
 24     return Promise.reject(`Failed to execute rsync: user is ${userError}`);
 25 
 26   let hostError = VALIDATE.IsStringInput(host);
 27   if (hostError)
 28     return Promise.reject(`Failed to execute rsync: host is ${hostError}`);
 29 
 30   let flagError = VALIDATE.IsStringInput(flag);
 31   if (flagError)
 32     return Promise.reject(`Failed to execute rsync: flag is ${flagError}`);
 33 
 34   let srcError = VALIDATE.IsStringInput(src);
 35   if (srcError)
 36     return Promise.reject(`Failed to execute rsync: source is ${srcError}`);
 37 
 38   let destError = VALIDATE.IsStringInput(dest);
 39   if (destError)
 40     return Promise.reject(`Failed to execute rsync: destination is ${destError}`);
 41 
 42   if (!executor)
 43     return Promise.reject(`Failed to execute rsync: Executor is required`);
 44 
 45   return new Promise((resolve, reject) => {
 46     executor.Execute('rsync', ['-a'].concat(flag).concat([src, `${user}@${host}:${dest}`])).then(output => {
 47       if (output.stderr) {
 48         reject(`Failed to execute rsync: ${output.stderr}`);
 49         return;
 50       }
 51       resolve(output.stdout);
 52     }).catch(error => reject(`Failed to execute rsync: ${error}`));
 53   });
 54 }
 55 
 56 /**
 57  * Sync files to host.
 58  * @param {string} user Username
 59  * @param {string} host Host name
 60  * @param {string} src Source
 61  * @param {string} dest Destination
 62  * @param {Command} executor Command object that will execute the command.
 63  * @returns {Promise<string>} Returns a promise. If it resolves, it returns a string with output message. Else, it rejects and returns an error.
 64  */
 65 function Rsync(user, host, src, dest, executor) {
 66   return Rsync_(user, host, [], src, dest, executor);
 67 }

 69 /**
 70  * Update files on host that were updated at the source. (Skips new files)
 71  * @param {string} user Username
 72  * @param {string} host Host name
 73  * @param {string} src Source
 74  * @param {string} dest Destination
 75  * @param {Command} executor Command object that will execute the command.
 76  * @returns {Promise<string>} Returns a promise. If it resolves, it returns a string with output message. Else, it rejects and returns an error.
 77  */
 78 function Update(user, host, src, dest, executor) { // Update dest if src was updated
 79   return Rsync_(user, host, '--update', src, dest, executor);
 80 }
 81 
 82 /**
 83  * Match files on host to those in the source. (Will delete files in host that do not exist in source).
 84  * @param {string} user Username
 85  * @param {string} host Host name
 86  * @param {string} src Source
 87  * @param {string} dest Destination
 88  * @param {Command} executor Command object that will execute the command.
 89  * @returns {Promise<string>} Returns a promise. If it resolves, it returns a string with output message. Else, it rejects and returns an error.
 90  */
 91 function Match(user, host, src, dest, executor) { // Copy files and then delete those NOT in src (Match dest to src)
 92   return Rsync_(user, host, '--delete-after', src, dest, executor);
 93 }
 94 
 95 /**
 96  * Execute rsync command using specified arguments.
 97  * @param {Array<string|number>} args List of args used in 'rsync' command.
 98  * @param {Command} executor Command object that will execute the command.
 99  * @returns {Promise<string>} Returns a promise. If it resolves, it returns a string with output message. Else, it rejects and returns an error.
100  */
101 function Manual(args, executor) {
102   let argsError = argsValidator(args);
103   if (argsError)
104     return Promise.reject(`Failed to execute rsync: ${argsError}`);
105 
106   if (!executor)
107     return Promise.reject(`Failed to execute rsync: Executor is required`);
108 
109   return new Promise((resolve, reject) => {
110     executor.Execute('rsync', args).then(output => {
111       if (output.stderr) {
112         reject(`Failed to execute rsync: ${output.stderr}`);
113         return;
114       }
115       resolve(output.stdout);
116     }).catch(error => reject(`Failed to execute rsync: ${error}`));
117   });
118 }

120 /**
121  * Run rsync command without affecting files. (For testing purposes).
122  * @param {Array<string|number>} args List of args used in 'rsync' command.
123  * @param {Command} executor Command object that will execute the command.
124  * @returns {Promise<string>} Returns a promise. If it resolves, it returns a string with output message. Else, it rejects and returns an error.
125  */
126 function DryRun(args, executor) { // Will execute without making changes (for testing command)
127   let argsError = argsValidator(args);
128   if (argsError)
129     return Promise.reject(`Failed to execute rsync: ${argsError}`);
130 
131   if (!executor)
132     return Promise.reject(`Failed to execute rsync: Executor is required`);
133 
134   return Manual(['--dry-run'].concat(args), executor);
135 }
