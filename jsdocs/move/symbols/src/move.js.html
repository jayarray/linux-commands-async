1 let VALIDATE = require('./validate.js');
  2 
  3 //------------------------------------------------------
  4 // MOVE 
  5 
  6 /**
  7  * Move a file or directory to another location.
  8  * @param {string} src Source
  9  * @param {string} dest Destination
 10  * @param {Command} executor Command object that will execute the command.
 11  * @returns {Promise} Returns a promise that resolves if successful, otherwise it returns an error.
 12  */
 13 function Move(src, dest, executor) {
 14   let srcError = VALIDATE.IsStringInput(src);
 15   if (srcError)
 16     return Promise.reject(`Failed to move: Source is ${srcError}`);
 17 
 18   let destError = VALIDATE.IsStringInput(dest);
 19   if (destError)
 20     return Promise.reject(`Failed to move: Destination is ${destError}`);
 21 
 22   if (!executor)
 23     return Promise.reject(`Failed to move: Executor is required`);
 24 
 25   return new Promise((resolve, reject) => {
 26     executor.Execute('mv', [src, dest]).then(output => {
 27       if (output.stderr) {
 28         reject(`Failed to move: ${output.stderr}`);
 29         return;
 30       }
 31       resolve();
 32     }).catch(error => reject(`Failed to move: ${error}`));
 33   });
 34 }
 35 
 36 //----------------------------------
 37 // EXPORTS
 38 
 39 exports.Move = Move;
