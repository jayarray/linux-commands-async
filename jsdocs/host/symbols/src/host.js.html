  5 /**
  6  * Get info about current host.
  7  * @param {Command} executor Command object that will execute the command.
  8  * @returns {Promise<{hostname: string, ipaddress: string}>} Returns a promise. If it resolves, it returns an object. Else, it returns an error.
  9  */
 10 function Info(executor) {
 11   if (!executor) {
 12     return Promise.reject(`Failed to get host info: Executor is required`);
 13   }
 14 
 15   return new Promise((resolve, reject) => {
 16     let cmdList = ['hostname', 'hostname -i'];
 17 
 18     executor.Execute(cmdList.join('\n'), []).then(output => {
 19       if (output.stderr) {
 20         reject(`Failed to get host info: ${output.stderr}`);
 21         return;
 22       }
 23 
 24       let lines = output.stdout.trim().split()
 25         .filter(line => line && line != '' && line.trim() != '')
 26         .map(line => line.trim());
 27 
 28       resolve({
 29         hostname: lines[0],
 30         ipaddress: lines[1]
 31       });
 32     }).catch(error => `Failed to get host info: ${error}`);
 33   });
 34 }
 35 
 36 //-------------------------------
 37 // EXPORTS
 38 
 39 exports.Info = Info;
