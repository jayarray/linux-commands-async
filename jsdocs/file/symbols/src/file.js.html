1 let VALIDATE = require('./validate.js');

  9 /**
 10  * Delete a file.
 11  * @param {string} path
 12  * @param {Command} executor Command object that will execute the command.
 13  * @returns {Promise} Returns a promise that resolves if successful, otherwise it returns an error.
 14  */
 15 function Remove(path, executor) {
 16   return REMOVE.Files([path], executor);
 17 }
 18 
 19 /**
 20  * Create a file.
 21  * @param {string} path
 22  * @param {string} text
 23  * @param {Command} executor Command object that will execute the command.
 24  * @returns {Promise} Returns a promise that will resolve if successful, otherwise it returns an error.
 25  */
 26 function Create(path, text, executor) {
 27   let pathError = VALIDATE.IsStringInput(path);
 28   if (pathError)
 29     return Promise.reject(`Failed to create file: path is ${pathError}`);
 30 
 31   let textError = VALIDATE.IsStringInput(text);
 32   if (textError)
 33     return Promise.reject(`Failed to create file: text is ${textError}`);
 34 
 35   if (!executor)
 36     return Promise.reject(`Failed to create file: Executor is required`);
 37 
 38   return new Promise((resolve, reject) => {
 39     executor.Execute(`echo "${text}" > ${path}`, []).then(output => {
 40       if (output.stderr) {
 41         reject(`Failed to create file: ${output.stderr}`);
 42         return;
 43       }
 44       resolve();
 45     }).catch(error => reject(`Failed to create file: ${error}`));
 46   });
 47 }

 49 /**
 50  * Make file executable.
 51  * @param {string} path
 52  * @param {Command} executor Command object that will execute the command.
 53  * @returns {Promise} Returns a promise that will resolve if successful, otherwise it returns an error.
 54  */
 55 function MakeExecutable(path, executor) {
 56   let pathError = VALIDATE.IsStringInput(path);
 57   if (pathError)
 58     return Promise.reject(`Failed to make file executable: path is ${pathError}`);
 59 
 60   if (!executor)
 61     return Promise.reject(`Failed to make file executable: Executor is required`);
 62 
 63   return new Promise((resolve, reject) => {
 64     PATH.IsFile(path, executor).then(isFile => {
 65       if (!isFile) {
 66         reject(`Failed to make file executable: path is not a file: ${path}`);
 67         return;
 68       }
 69 
 70       CHMOD.AddPermissions('ugo', 'x', [path], false, executor).then(success => {
 71         resolve();
 72       }).catch(error => reject(`Failed to make file executable: ${error}`));
 73     }).catch(error => reject(`Failed to make file executable: ${error}`));
 74   });
 75 }

 77 /**
 78  * Retrieve file content as a single string.
 79  * @param {string} path
 80  * @param {Command} executor Command object that will execute the command.
 81  * @returns {Promise<string>} Returns a promise. If it resolves, it returns a string. Else, it returns an error.
 82  */
 83 function Read(path, executor) {
 84   let pathError = VALIDATE.IsStringInput(path);
 85   if (pathError)
 86     return Promise.reject(`Failed to read file: path is ${pathError}`);
 87 
 88   if (!executor)
 89     return Promise.reject(`Failed to read file: Executor is required`);
 90 
 91   return new Promise((resolve, reject) => {
 92     PATH.IsFile(path, executor).then(isFile => {
 93       if (!isFile) {
 94         reject(`Failed to read file: path is not a file: ${path}`);
 95         return;
 96       }
 97 
 98       executor.Execute('cat', [path]).then(output => {
 99         if (output.stderr) {
100           reject(`Failed to read file: ${output.stderr}`);
101           return;
102         }
103         resolve(output.stdout);
104       }).catch(error => reject(`Failed to read file: ${error}`));
105     }).catch(error => reject(`Failed to read file: ${error}`));
106   });
107 }
108 
109 /**
110  * Retrieve file content as multiple strings.
111  * @param {string} path
112  * @param {Command} executor Command object that will execute the command.
113  * @returns {Promise<string>} Returns a promise. If it resolves, it returns an array of strings representing lines of text. Else, it returns an error.
114  */
115 function ReadLines(path, executor) {
116   return new Promise((resolve, reject) => {
117     Read(path, executor).then(string => {
118       resolve(string.split('\n'));
119     }).catch(error => reject(`Failed to read lines: ${error}`));
120   });
121 }
