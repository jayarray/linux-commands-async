1 let VALIDATE = require('./validate.js');
  2 
  3 //-------------------------------------------------
  4 // REMOVE (rm)
  5 
  6 /**
  7  * Delete a file.
  8  * @param {Array<string>} paths List of filepaths to delete.
  9  * @param {Command} executor Command object that will execute the command.
 10  * @returns {Promise} Returns a promise that resolves if successful, otherwise rejects and returns an error.
 11  */
 12 function Files(paths, executor) {
 13   let pathsError = VALIDATE.IsArray(paths);
 14   if (pathsError)
 15     return Promise.reject(`Failed to remove files: paths are ${pathsError}`);
 16 
 17   if (!executor)
 18     return Promise.reject(`Failed to remove files: Executor is required`);
 19 
 20   return new Promise((resolve, reject) => {
 21     executor.Execute('rm', ['-f'].concat(paths)).then(output => {
 22       if (output.stderr) {
 23         reject(`Failed to remove files: ${output.stderr}`);
 24         return;
 25       }
 26       resolve();
 27     }).catch(error => reject(`Failed to remove files: ${error}`));
 28   });
 29 }
 30
