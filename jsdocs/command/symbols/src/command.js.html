1 const CHILD_PROCESS = require('child_process');
  2 const VALIDATE = require('./validate.js');

  5 // HELPERS
  6 
  7 function argsValidator(args) {
  8   let argsError = VALIDATE.IsArray(args);
  9   if (argsError)
 10     return `arguments are ${argsError}`;
 11 
 12   for (let i = 0; i < args.length; ++i) {
 13     let currArg = args[i];
 14 
 15     let stringError = VALIDATE.IsStringInput(currArg);
 16     let numberError = VALIDATE.IsNumber(currArg);
 17 
 18     if (stringError && numberError)
 19       return `arguments must be string or number type`;
 20   }
 21   return null;
 22 }

 24 //---------------------------------------------
 25 // SAVING DATA (to string)
 26 
 27 class SavedData {
 28   constructor(stream) {
 29     this.value_ = '';
 30     stream.on('data', this.Callback_.bind(this));
 31   }
 32 
 33   Callback_(data) {
 34     this.value_ += data.toString();
 35   }
 36 
 37   get value() {
 38     return this.value_;
 39   }
 40 }

 42 //-----------------------------------------
 43 // LOCAL
 44 
 45 class Command {
 46   constructor() {
 47   }
 48 
 49   /**
 50    * Executes a system command locally.
 51    * @param {string} cmd Command name or command string.
 52    * @param {Array<string|number>} args List of args associated with command name. (Assign as empty array if no args are needed).
 53    * @returns {Promise<{stderr: string, stdout: string, exitCode: number}>} Returns a promise. If it resolves, it returns an object. Else, it returns an error.
 54    */
 55   Execute(cmd, args) {
 56     let cmdError = VALIDATE.IsStringInput(cmd);
 57     if (cmdError)
 58       return Promise.reject(`Failed to execute command: command is ${cmdError}`);
 59 
 60     let argsError = argsValidator(args);
 61     if (argsError)
 62       return Promise.reject(`Failed to execute command: ${argsError}`);
 63 
 64     return new Promise((resolve) => {
 65       let childProcess = null;
 66 
 67       if (args.length > 0)
 68         childProcess = childProcess = CHILD_PROCESS.spawn(cmd, args);
 69       else
 70         childProcess = CHILD_PROCESS.exec(cmd);
 71 
 72       let stderr = new SavedData(childProcess.stderr);
 73       let stdout = new SavedData(childProcess.stdout);
 74 
 75       childProcess.on('close', exitCode => {
 76         resolve({
 77           stderr: stderr.value,
 78           stdout: stdout.value,
 79           exitCode: exitCode
 80         });
 81       });
 82     });
 83   }
 84 }

 89 // REMOTE
 90 
 91 class RemoteCommand extends Command {
 92   /**
 93   * @param {string} user 
 94   * @param {string} host 
 95   */
 96   constructor(user, host) {
 97     super();
 98     this.user_ = user;
 99     this.host_ = host;
100   }
101 
102   /**
103    * @override
104    * Executes a system command remotely.
105    * @param {string} cmd Command name or command string.
106    * @param {Array<string|number>} args List of args associated with command name. (Assign as empty array if no args are needed).
107    * @returns {Promise<{stderr: string, stdout: string, exitCode: number}>} Returns a promise. If it resolves, it returns an object. Else, it rejects and returns an error.
108    */
109   Execute(cmd, args) {
110     let cmdError = VALIDATE.IsStringInput(cmd);
111     if (cmdError)
112       return Promise.reject(`Failed to execute command: command is ${cmdError}`);
113 
114     let argsError = argsValidator(args);
115     if (argsError)
116       return Promise.reject(`Failed to execute command: ${argsError}`);
117 
118     let sshArgs = [`${this.user_}@${this.host_}`];
119 
120     if (args.length > 0)
121       sshArgs.push(`${cmd} ${args.join(' ')}`);
122     else
123       sshArgs.push(cmd);
124 
125     return LOCAL.Execute('ssh', sshArgs);
126   }
127 
128   /**
129    * Creates a Command object that executes commands remotely.
130    * @param {string} user Username
131    * @param {string} host Remote host name
132    * @returns {Promise<Command>} Returns a promise. If it resolves, it returns a Command object. Else, it returns an error.
133    */
134   static Create(user, host) {
135     let userError = VALIDATE.IsStringInput(user);
136     if (userError)
137       return Promise.reject(userError);
138 
139     let hostError = VALIDATE.IsStringInput(host);
140     if (hostError)
141       return Promise.reject(hostError);
142 
143     return Promise.resolve(new RemoteCommand(user, host));
144   }
145 }
