  7 /**
  8  * Check the status of multiple paths.
  9  * @param {Array<string>} paths List of paths to query.
 10  * @param {Command} executor Command object that will execute the command.
 11  * @returns {Promise<{path: string}>} Returns a promise. If it resolves, it returns an dictionary of all queried paths, each of which is assigned one of the following strings: 'f' (file), 'd' (directory), 'dne' (does not exist), 'invalid'. Else, it rejects and returns an error.
 12  */
 13 function Query(paths, executor) {
 14   let pathsError = PathsValidator(paths);
 15   if (pathsError)
 16     return Promise.reject(`Failed to determine if paths are files or directories: ${pathsError}`);
 17 
 18   if (!executor)
 19     return Promise.reject(`Failed to determine if paths are files or directories: Executor is required`);
 20 
 21   return new Promise((resolve, reject) => {
 22     let cmdList = [];
 23     paths.forEach(path => {
 24       let cmd = `if [ -e ${path} ]; then`; // can you do !exists and return dne to avoid nested ifs?
 25       cmd += ` if [ -d ${path} ]; then echo "d";`;
 26       cmd += ` else`;
 27       cmd += ` if [ -f ${path} ]; then  echo "f";`;
 28       cmd += ` else echo "invalid";`;
 29       cmd += ` fi fi`;
 30       cmd += ` else echo "dne"; fi`;
 31 
 32       cmdList.push(cmd);
 33     });
 34 
 35     executor.Execute(cmdList.join('\n'), []).then(output => {
 36       if (output.stderr) {
 37         reject(`Failed to determine if paths are files or directories: ${outptu.stderr}`);
 38         return;
 39       }
 40 
 41       let lines = output.stdout.trim().split('\n')
 42         .filter(line => line && line != '' && line.trim() != '')
 43         .map(line => line.trim());
 44 
 45       let dict = {};
 46       for (let i = 0; i < lines.length; ++i) {
 47         let currLine = lines[i];
 48         let currPath = paths[i];
 49         dict[currPath] = currLine;
 50       }
 51 
 52       resolve(dict);
 53     }).catch(error => `Failed to determine if paths are files or directories: ${error}`);
 54   });
 55 }

 57 /**
 58  * Check if path exists.
 59  * @param {string} path
 60  * @param {Command} executor Command object that will execute the command.
 61  * @returns {Promise<boolean>} Returns a promise. If it resolves, it returns a boolean value. Else, it returns an error.
 62  */
 63 function Exists(path, executor) {
 64   let pathError = PathValidator(path);
 65   if (pathError)
 66     return Promise.reject(`Failed to check if path exists: ${pathError}`);
 67 
 68   if (!executor)
 69     return Promise.reject(`Failed to check if path exists: Executor is required`);
 70 
 71   return new Promise((resolve, reject) => {
 72     Query([path], executor).then(dict => {
 73       let type = dict[path];
 74       if (type == 'f' || type == 'd')
 75         resolve(true);
 76       else
 77         resolve(false);
 78     }).catch(error => reject(`Failed to check if path is a file: ${error}`));
 79   });
 80 }

167 // HELPERS

169 function PathValidator(p) {
170   let error = VALIDATE.IsStringInput(p);
171   if (error)
172     return `Path is ${error}`;
173   return null;
174 }
175 
176 function PathsValidator(paths) {
177   let error = VALIDATE.IsArray(paths);
178   if (error)
179     return `paths are ${error}`;
180 
181   for (let i = 0; i < paths.length; ++i) {
182     let invalidType = VALIDATE.IsStringInput(paths[i]);
183     if (invalidType)
184       return `paths contain a path that is ${invalidType}`;
185   }
186   return null;
187 }
