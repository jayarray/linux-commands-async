  7 /**
  8  * List all files and directories (visible and hidden) with their respective sizes.
  9  * @param {string} dirPath Directory location.
 10  * @param {Command} executor Command object that will execute the command.
 11  * @returns {Promise<Array<{size: number, path: string}>>} Returns a promise. If it resolves, it returns an array of objects. Else, it and returns an error.
 12  */
 13 function ListAllItems(dirPath, executor) {
 14   let dirPathError = VALIDATE.IsStringInput(dirPath);
 15   if (dirPathError)
 16     return Promise.reject(`Failed to list all disk usage items: Directory path is ${dirPathError}`);
 17 
 18   if (!executor)
 19     return Promise.reject(`Failed to list all disk usage items: Executor is required`);
 20 
 21   return new Promise((resolve, reject) => {
 22     PATH.IsDir(dirPath, executor).then(isDir => {
 23       if (!isDir) {
 24         reject(`Failed to list all disk usage items: Path is not a directory: ${dirPath}`);
 25         return;
 26       }
 27 
 28       executor.Execute('du', ['-ha', '--block-size=1', '--max-depth=1', dirPath]).then(output => {
 29         if (output.stderr) {
 30           reject(`Failed to list all disk usage items: ${output.stderr}`);
 31           return;
 32         }
 33 
 34         let items = [];
 35 
 36         let lines = output.stdout.trim().split('\n').map(line => line.trim()).filter(line => line && line != '');
 37         lines.forEach(line => {
 38           let sizeStr = '';
 39           for (let i = 0; i < line.length; ++i) {
 40             let currChar = line.charAt(i);
 41             if (currChar.trim())
 42               sizeStr += currChar;
 43             else
 44               break;
 45           }
 46 
 47           let filepath = line.substring(sizeStr.length + 1);
 48           items.push({ size: parseInt(sizeStr), path: filepath });
 49         });
 50         resolve(items.filter(item => item.path != dirPath));
 51       }).catch(error => reject(`Failed to list all disk usage items: ${error}`));
 52     }).catch(error => reject(`Failed to list all disk usage items: ${error}`));
 53   });
 54 }

 56 /**
 57  * List all visible files and directories with their respective sizes.
 58  * @param {string} dirPath Directory location.
 59  * @param {Command} executor Command object that will execute the command.
 60  * @returns {Promise<Array<{size: number, path: string}>>} Returns a promise. If it resolves, it returns an array of objects. Else, it and returns an error.
 61  */
 62 function ListVisibleItems(dirPath, executor) {
 63   return new Promise((resolve, reject) => {
 64     ListAllItems(dirPath, executor).then(items => {
 65       resolve(items.filter(item => !PATH.Filename(item.path).startsWith('.')));
 66     }).catch(error => reject(`Failed to list visible disk usage items: ${error}`));
 67   });
 68 }

 84 /**
 85  * Determine folder size.
 86  * @param {string} dirPath Directory location.
 87  * @param {Command} executor Command object that will execute the command.
 88  * @returns {Promise<number>} Returns a promise. If it resolves, it returns an integer representing the size in bytes. Else, it rejects and returns an error.
 89  */
 90 function DirSize(dirPath, executor) {
 91   let dirPathError = VALIDATE.IsStringInput(dirPath);
 92   if (dirPathError)
 93     return Promise.reject(`Failed to get directory size: Directory path is ${dirPathError}`);
 94 
 95   if (!executor)
 96     return Promise.reject(`Failed to get directory size: Executor is required`);
 97 
 98   return new Promise((resolve, reject) => {
 99     PATH.IsDir(dirPath, executor).then(isDir => {
100       if (!isDir) {
101         reject(`Failed to get directory size: Path is not a directory: ${dirPath}`);
102         return;
103       }
104 
105       executor.Execute('du', ['-sh', '--block-size=1', dirPath]).then(output => {
106         if (output.stderr) {
107           reject(`Failed to get directory size: ${output.stderr}`);
108           return;
109         }
110 
111         let outputStr = output.stdout.trim();
112 
113         let sizeStr = '';
114         for (let i = 0; i < outputStr.length; ++i) {
115           let currChar = outputStr.charAt(i);
116           if (currChar.trim())
117             sizeStr += currChar;
118           else
119             break;
120         }
121         resolve(parseInt(sizeStr));
122       }).catch(error => reject(`Failed to get directory size: Directory path is ${error}`));
123     }).catch(error => reject(`Failed to get directory size: Directory path is ${error}`));
124   });
125 }
