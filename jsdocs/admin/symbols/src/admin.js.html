  9 function parseUptimeString(string) {
 10   // System info
 11   let firstParts = string.split(',')[0].split(' ')
 12     .filter(part => part && part != '' && part.trim() != '')
 13     .map(part => part.trim());
 14 
 15   let timestamp = firstParts[0];
 16   let uptime = firstParts.slice(2).join(' ');
 17 
 18   // Number of users
 19   let secondParts = string.split(',')[1].split(' ')
 20     .filter(part => part && part != '' && part.trim() != '')
 21     .map(part => part.trim());
 22   let userCount = parseInt(secondParts[0]);
 23 
 24   let thirdParts = string.split(',')
 25     .slice(2).join(',')
 26     .split(':')[1]
 27     .split(',')
 28     .filter(part => part && part != '' && part.trim() != '')
 29     .map(part => part.trim());
 30 
 31   // Load averages
 32   let loadAverages = [];
 33   thirdParts.forEach(part => loadAverages.push(parseFloat(part)));
 34 
 35   let obj = {
 36     timestamp: timestamp,
 37     uptime: uptime,
 38     userCount: userCount,
 39     loadAverages: loadAverages
 40   };
 41   return obj;
 42 }

 46 function getHeaders(string) {
 47   let headers = {};
 48 
 49   let startIndex = 0;
 50   let endIndex = 0;
 51   let char = string.charAt(endIndex);
 52   let str = '';
 53 
 54   while (startIndex < string.length && endIndex < string.length) {
 55     if (char.trim() == '') {
 56       let strTrimmed = str.trim();
 57       if (strTrimmed) {
 58         headers[string.substring(startIndex, endIndex)] = { start: startIndex, end: endIndex - 1 };
 59         startIndex = endIndex;
 60         str = '';
 61       }
 62       else {
 63         startIndex += 1;
 64         endIndex += 1;
 65       }
 66     }
 67     else {
 68       str += char;
 69       endIndex += 1;
 70     }
 71     char = string.charAt(endIndex);
 72   }
 73 
 74   if (str)
 75     headers[string.substring(startIndex, endIndex)] = { start: startIndex, end: endIndex - 1 };
 76   return headers;
 77 }

 94 function getStringFromEndIndex(line, endIndex) {
 95   let startIndex = endIndex;
 96 
 97   let char = line.charAt(startIndex);
 98   while (char.trim() != '') {
 99     startIndex -= 1;
100     char = line.charAt(startIndex);
101   }
102   return line.substring(startIndex, endIndex + 1);
103 }

165 function getLsofObject(line, headers) {

167 
168   let parts = line.split(' ')
169     .filter(part => part && part != '' && part.trim() != '')
170     .map(part => part.trim());
171 
172   let partIndex = 0;
173   let lsofObj = {};
174 
175   Object.keys(headers).forEach(field => {
176     let name = field;
177     if (field.includes('/'))
178       name = field.replace('/', '');
179     name = name.toLocaleLowerCase();
180 
181     if (hasValues[field]) {
182       if (field == 'COMMAND' ||
183         field == 'PID' ||
184         field == 'USER' ||
185         field == 'FD' ||
186         field == 'TYPE' ||
187         field == 'DEVICE' ||
188         field == 'SIZE/OFF' ||
189         field == 'NODE') {
190         if (isNaN(field))
191           lsofObj[name] = parts[partIndex];
192         else
193           lsofObj[name] = parseInt(parts[partIndex]);
194       }
195       else if (field == 'NAME') {
196         lsofObj[name] = parts.slice(partIndex).join(' ');
197       }
198     }
199     else
200       lsofObj[name] = null;
201 
202     partIndex += 1;
203   });
204   return lsofObj;

369 function Processes(executor) { // Gives status of running processes with a unique id called PID and other fields.
370   if (!executor)
371     return Promise.reject(`Failed to get running processes: Executor is required`);
372 
373   return new Promise((resolve, reject) => {
374     let fields = ['user', 'uid', 'gid', 'tty', 'start', 'etime', 'pid', 'ppid', 'pgid', 'tid', 'tgid', 'sid', 'ruid', 'rgid', 'euid', 'suid', 'fsuid', 'comm'];
375 
376     executor.Execute('ps', ['xao', fields.join(',')]).then(output => {
377       if (output.stderr) {
378         reject(`Failed to get running processes: ${output.stderr}`);
379         return;
380       }
381 
382       // Get lines
383       let lines = output.stdout.trim().split('\n')
384         .filter(line => line && line != '' && line.trim() != '')
385         .map(line => line.trim());
386 
387       // Get headers
388       let headers = getHeaders(lines[0]);
389 
390       // Parse lines into process objects
391       let processes = [];
392 
393       lines.slice(1).forEach(line => {
394         let parts = line.split(' ')
395           .filter(part => part && part != '' && part.trim() != '')
396           .map(part => part.trim());
397 
398         let process = {};
399         for (let i = 0; i < fields.length; ++i) {
400           let currField = fields[i];
401           let currValue = parts[i];
402           if (currField.endsWith('id'))
403             process[currField] = parseInt(currValue);
404           else
405             process[currField] = currValue;
406         }
407         processes.push(process);
408       });
409       resolve(processes);
410     }).catch(error => reject(`Failed to get running processes: ${error}`));
411   });
412 }

414 /**
415  * Retrieve the running process that matches the specified process id.
416  * @param {number} pid Process ID
417  * @param {Command} executor Command object that will execute the command.
418  * @returns {Promise<{user: string, uid: number, gid: number, tty: string, start: string, etime: string, pid: number, ppid: number, pgid: number, tid: number, tgid: number, sid: number, ruid: number, rgid: number, euid: number, suid: number, fsuid: number, comm: string}>} Returns a promise. If it resolves, it returns an object. Else, it returns an error.
419  */
420 function GetProcess(pid, executor) {
421   let pidError = VALIDATE.IsInteger(pid);
422   if (pidError)
423     return Promise.reject(`Failed to get process: pid is ${pidError}`);
424 
425   let min = 0;
426   let boundError = VALIDATE.IsIntegerInRange(pid, min, null);
427   if (boundError)
428     return Promise.reject(`Failed to get process: ${boundError}`);
429 
430   if (!executor)
431     return Promise.reject(`Failed to get process: Executor is required`);
432 
433   return new Promise((resolve, reject) => {
434     Processes(executor).then(processes => {
435       for (let i = 0; i < processes.length; ++i) {
436         let currProcess = processes[i];
437         if (currProcess.pid == pid) {
438           resolve(currProcess);
439           return;
440         }
441       }
442       reject(`Failed to get process: process does not exist: ${pid}`);
443     }).catch(error => reject(`Failed to get process: ${error}`));
444   });
445 }

516     return Promise.reject(`Failed to get top processes: Executor is required`);
517

521         reject(`Failed to get top processes: ${output.stderr}`);
522         return;
523       }
524 
525       let lines = output.stdout.split('\n')
526         .filter(line => line && line != '' && line.trim() != '')
527         .map(line => line.trim());
528 
529       // Uptime str
530       let uptimeStr = lines[0].replace('top -', '');
531       let uptimeObj = parseUptimeString(uptimeStr);
532 
533       // Tasks str
534       let tasksStr = lines[1].replace('Tasks:', '').trim();
535       let taskParts = tasksStr.split(',').map(part => part.trim());
536 
537       let tasks = {};
538       taskParts.forEach(part => {
539         let parts = part.split(' ').map(p => p.trim());
540         let int = parseInt(parts[0].trim());
541         let name = parts[1].trim();
542         tasks[name] = int;
543       });
544 
545       // cpu str
546       let cpuStr = lines[2].replace('%Cpu(s):', '');
547       let cpuParts = cpuStr.split(',').map(part => part.trim());
548 
549       let cpu = {};
550       cpuParts.forEach(part => {
551         let parts = part.split(' ').map(p => p.trim());
552         let float = parseFloat(parts[0].trim());
553         let name = parts[1].trim();
554         cpu[name] = float
555       });
556 
557       // mem str
558       let memStr = lines[3].split(':')[1].trim();
559       let memParts = memStr.split(',').map(part => part.trim());
560 
561       let mem = {};
562       memParts.forEach(part => {
563         let parts = part.split(' ').map(p => p.trim());
564         let float = parseInt(parts[0].trim());
565 
566         let name = parts[1].trim();
567         if (name.includes('/'))
568           name = name.replace('/', '');
569 
570         mem[name] = float
571       });
572 
573       // swap str
574       let swapStr = lines[4].split(':')[1].trim();
575       let swapParts = swapStr.split(',').map(part => part.trim());
576

578       for (let i = 0; i < 2; ++i) {
579         let currSwapPart = swapParts[i];

581         let int = parseInt(parts[0].trim());
582         let name = parts[1].trim();

584       }
585 
586       let otherParts = swapParts[2].split('.');
587

591       let int = parseInt(parts[0].trim());
592       let name = parts[1].trim();
593       swap[name] = int;
594 
595       // swap: avail Mem
596       currPart = otherParts[1].trim();
597       parts = currPart.split(' ');
598       int = parseInt(parts[0].trim());
599       let firstWord = parts[1].trim();
600       let secondWord = parts[2].trim().toLowerCase();
601       name = firstWord + secondWord;
602       swap[name] = int;
603 
604       // headers
605       let headersStr = lines[5].trim();
606       let headers = getHeaders(headersStr);
607 
608       // rows
609       let processes = [];
610 
611       lines.slice(6).forEach(line => {

613           .filter(part => part && part != '' && part.trim() != '')
614           .map(part => part.trim());
615

617         let user = parts[1];
618         let pr = parseInt(parts[2]);
619         let ni = parseInt(parts[3]);
620         let virt = parseInt(parts[4]);
621         let res = parseInt(parts[5]);
622         let shr = parseInt(parts[6]);
623         let s = parts[7];
624         let cpu = parseFloat(parts[8]);
625         let mem = parseFloat(parts[9]);
626         let time = parts[10];
627         let command = parts[11]

631           user: user,
632           pr: pr,

634           virt: virt,
635           res: res,
636           shr: shr,
637           s: s,
638           cpu: cpu,
639           mem: mem,
640           time: time,
641           command: command
642         };
643         processes.push(process);
644       });
645 
646       let result = {
647         uptime: uptimeObj,
648         tasks: tasks,
649         cpu: cpu,
650         mem: mem,
651         swap: swap,
652         processes: processes
653       };
654 
655       resolve(result);
656     }).catch(error => reject(`Failed to get top processes: ${error}`));
657   });

676       let lines = output.stdout.trim().split('\n')
677         .filter(line => line && line != '' && line.trim() != '')
678         .map(line => line.trim());
679 
680       // Uptime info
681       let uptimeStr = lines[0];
682       let uptimeObj = parseUptimeString(uptimeStr);
683 
684       // Headers
685       let headersLine = lines[1];
686       let headers = getHeaders(headersLine);
687 
688       // User info
689       let users = [];
690 
691       let userLines = lines.slice(2)
692         .filter(line => line && line != '' && line.trim() != '')
693         .map(line => line.trim());
694 
695       userLines.forEach(line => {
696         let parts = line.split(' ')
697           .filter(part => part && part != '' && part.trim() != '')
698           .map(part => part.trim());
699 
700         let name = parts[0];
701         let tty = parts[1];
702         let group = parts[2];
703         let login = parts[3];
704         let idle = parts[4];
705         let jcpu = parts[5];
706         let pcpu = parts[6];
707         let what = parts.slice(7).join(' ');
708 
709         let user = {
710           name: name,
711           tty: tty,
712           group: group,
713           login: login,
714           idle: idle,
715           jcpu: jcpu,
716           pcpu: pcpu,
717           what: what,
718         }
719         users.push(user);
720       });
721       resolve({
722         users: users,
723         uptimeObj: uptimeObj
724       });
725     }).catch(error => reject(`Failed to get logged in users: ${error}`));
726   });
727 }

776  * @param {string|number} uid Username or user ID number.
777  * @param {Command} executor Command object that will execute the command.
778  * @returns {Promise<boolean>} Returns a promise. If it resolves, it returns a boolean value. Else, it returns an error.
779  */
780 function UserHasRootPermissions(uid, executor) {
781   let uidError = VALIDATE.IsInteger(uid);
782   if (uidError)
783     return Promise.reject(`Failed to verify if user has root permissions: pid is ${uidError}`);
784 
785   if (!executor)
786     return Promise.reject(`Failed to verify if user has root permissions: Executor is required`);
787 
788   return new Promise((resolve, reject) => {
789     GetUser(uid, executor).then(user => {
790       if (user.name == 'root') {
791         resolve(true);
792         return;
793       }
794 
795       GetGroup('root', executor).then(group => {
796         resolve(group.users.includes(user.name));
797       }).catch(error => reject(`Failed to verify if user has root permissions: ${error}`));
798     }).catch(error => reject(`Failed to verify if user has root permissions: ${error}`));
799   });
800 }

803  * Check if user has permissions to change group ownership.
804  * @param {string|number} uid Username or user ID number.
805  * @param {string|number} desiredGid Group name or group ID number.
806  * @param {Command} executor Command object that will execute the command.
807  * @returns {Promise<boolean>} Returns a promise. If it resolves, it returns a boolean value. Else, it returns an error.
808  */
809 function UserCanChangeGroupOwnership(uid, desiredGid, executor) { // Must be root OR be part of the desired group to give ownership to that group.
810   let uidError = IdError(uid);
811   if (uidError)
812     return Promise.reject(`Failed to verify if user can change group ownership: uid ${uidError}`);
813 
814   let desiredGidError = IdError(desiredGid);
815   if (desiredGidError)
816     return Promise.reject(`Failed to verify if user can change group ownership: desired gid ${desiredGidError}`);
817 
818   if (!executor)
819     return Promise.reject(`Failed to verify if user can change group ownership: Executor is required`);
820 
821   return new Promise((resolve, reject) => {
822     GetUser(uid, executor).then(user => {
823       UserHasRootPermissions(user.id, executor).then(hasRootAccess => {
824         if (hasRootAccess) {
825           resolve(true);
826           return;
827         }
828 
829         GetGroup(desiredGid, executor).then(group => {
830           USERINFO.OtherUser(user.name, executor).then(info => {
831             let userGroupIds = info.groups.map(group => group.gid);
832             resolve(group.users.includes(user.name) || userGroupIds.includes(group.id));
833           }).catch(error => reject(`Failed to verify if user can change group ownership: ${error}`));
834         }).catch(error => reject(`Failed to verify if user can change group ownership: ${error}`));
835       }).catch(error => error => reject(`Failed to verify if user can change group ownership: ${error}`));
836     }).catch(error => error => reject(`Failed to verify if user can change group ownership: ${error}`));
837   });
838 }

859 function IdStringError(string) {
860   let error = VALIDATE.IsStringInput(string);
861   if (error)
862     return `is ${error} `;
863   return null;
864 }

866 function IdNumberError(int) {
867   let error = VALIDATE.IsInstance(int);
868   if (error)
869     return `is ${error} `;
870 
871   let min = 0;
872   error = VALIDATE.IsIntegerInRange(int, min, null);
873   if (error)
874     return error;
875   return null;
876 }
877 
878 function IdError(id) {
879   let error = VALIDATE.IsInstance(id);
880   if (error)
881     return `is ${error} `;
882 
883   // Check if string
884   if (typeof id == 'string') {
885     error = IdStringError(id);
886     if (error)
887       return error;
888     return null;
889   }
890 
891   // Check if number
892   if (Number.isInteger(id)) {
893     error = IdNumberError(id);
894     if (error)
895       return error;
896     return null;
897   }
898 
899   return `is not a valid string or integer`;
900 }
