  6 /**
  7  * Change user ownership.
  8  * @param {Array<string>} paths A list of paths that will change user ownership.
  9  * @param {string|Number} newOwnerId Username or user ID number.
 10  * @param {boolean} isRecursive Assign as true if ownership is to be applied recursively, otherwise assign as false.
 11  * @param {Command} executor Command object that will execute the command.
 12  * @returns {Promise} Returns a promise that will resolve if successful, otherwise it returns an error.
 13  */
 14 function ChangeOwner(paths, newOwnerId, isRecursive, executor) { // newOwner can be string or integer
 15   let pathsError = PathsValidator(paths);
 16   if (pathsError)
 17     return Promise.reject(`Failed to change owner: ${pathsError}`);
 18 
 19   let idError = NewIdValidator(newOwnerId);
 20   if (idError)
 21     return Promise.reject(`Failed to change owner: new owner id ${idError}`);
 22 
 23   if (!executor)
 24     return Promise.reject(`Failed to change owner: Executor is required`);
 25 
 26   return new Promise((resolve, reject) => {
 27     let args = [];
 28     if (isRecursive)
 29       args.push('-R');
 30     args.push(newOwnerId);
 31     args = args.concat(paths);
 32 
 33     executor.Execute('chown', args).then(output => {
 34       if (output.stderr) {
 35         reject(`Failed to change owner: ${output.stderr}`);
 36         return;
 37       }
 38       resolve();
 39     }).catch(error => reject(`Failed to change owner: ${error}`));
 40   });
 41 }

150 function NewIdValidator(id) {
151   let error = VALIDATE.IsInstance(id);
152   if (error)
153     return `is ${error}`;
154 
155   if (typeof id != 'string' && !Number.isInteger(id))
156     return `must be a string or integer`;
157 
158   return null;
159 }
160 
161 function PathsValidator(paths) {
162   let error = VALIDATE.IsArray(paths);
163   if (error)
164     return `Paths are ${error}`;
165 
166   for (let i = 0; i < paths.length; ++i) {
167     let currPath = paths[i];
168     let pathIsValid = VALIDATE.IsStringInput(currPath) == null;
169 
170     if (!pathIsValid)
171       return `All paths must be string type`;
172   }
173 
174   return null;
175 }
176 
177 function ArgsValidator(args) {
178   let error = VALIDATE.IsArray(args);
179   if (error)
180     return `arguments are ${error}`;
181 
182   for (let i = 0; i < args.length; ++i) {
183     let currArg = args[i];
184     let argIsValidString = VALIDATE.IsStringInput(currArg) == null;
185     let argIsValidNumber = !isNaN(currArg);
186 
187     if (!argIsValidString && !argIsValidNumber)
188       return `arg elements must be string or number type`;
189   }
190 
191   return null;
192 }
