1 let VALIDATE = require('./validate.js');
  2 
  3 //------------------------------------------------------
  4 // MKDIR (mkdir)
  5 
  6 /**
  7  * Create a directory.
  8  * @param {string} path
  9  * @param {Command} executor Command object that will execute the command.
 10  * @returns {Promise} Returns a promise that resolves if successful, otherwise it returns an error.
 11  */
 12 function MakeDirectory(path, executor) {
 13   let pathError = VALIDATE.IsStringInput(path);
 14   if (pathError)
 15     return Promise.reject(`Failed to make directory: ${pathError}`);
 16 
 17   if (!executor)
 18     return Promise.reject(`Failed to make directory: Executor is required`);
 19 
 20   return new Promise((resolve, reject) => {
 21     executor.Execute('mkdir', ['-p', path]).then(output => {
 22       if (output.stderr) {
 23         reject(`Failed to make directory: ${output.stderr}`);
 24         return;
 25       }
 26       resolve();
 27     }).catch(error => reject(`Failed to make directory: ${error}`));
 28   });
 29 }
 30 
 31 //-------------------------------------
 32 // EXPORTS
 33 
 34 exports.MakeDirectory = MakeDirectory;
