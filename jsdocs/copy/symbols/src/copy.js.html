  4 // COPY (cp)
  5 
  6 /**
  7  * Copy file.
  8  * @param {string} src Source
  9  * @param {string} dest Destination
 10  * @param {Command} executor Command object that will execute the command.
 11  * @returns {Promise} Returns a promise that will resolve if successful, otherwise returns an error.
 12  */
 13 function File(src, dest, executor) {
 14   let srcError = VALIDATE.IsStringInput(src);
 15   if (srcError)
 16     return Promise.reject(`Failed to copy file: Source is ${srcError}`);
 17 
 18   let destError = VALIDATE.IsStringInput(dest);
 19   if (destError)
 20     return Promise.reject(`Failed to copy file: Destination is ${destError}`);
 21 
 22   if (!executor)
 23     return Promise.reject(`Failed to copy file: Executor is required`);
 24 
 25   return new Promise((resolve, reject) => {
 26     executor.Execute('cp', [src, dest]).then(output => {
 27       if (output.stderr) {
 28         reject(`Failed to copy file: ${output.stderr}`);
 29         return;
 30       }
 31       resolve();
 32     }).catch(error => reject(`Failed to copy file: ${error}`));
 33   });
 34 }
