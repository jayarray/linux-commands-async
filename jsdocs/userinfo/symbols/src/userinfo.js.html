  6 /**
  7  * Get current user name.
  8  * @param {Command} executor Command object that will execute the command.
  9  * @returns {Promise<string>} Returns a promise. If it resolves, it returns a string. Else, it returns an error.
 10  */
 11 function WhoAmI(executor) {
 12   if (!executor)
 13     return Promise.reject(`Failed to identify who you are: Executor is required`);
 14 
 15   return new Promise((resolve, reject) => {
 16     executor.Execute('whoami', []).then(output => {
 17       if (output.stderr) {
 18         reject(`Failed to identify who you are: ${output.stderr}`);
 19         return;
 20       }
 21       resolve(output.stdout.trim());
 22     }).catch(error => reject(`Failed to identify who you are: ${error}`));
 23   });
 24 }

 26 /**
 27  * Get current user info.
 28  * @param {Command} executor Command object that will execute the command.
 29  * @returns {Promise<{username: string, uid: number, gid: number, groups: Array<{gid: number, name: string}> >}>} Returns a promise. If it resolves, it returns an object. Else, it returns an error.
 30  */
 31 function CurrentUser(executor) {
 32   if (!executor)
 33     return Promise.reject(`Failed to identify current user: Executor is required`);
 34 
 35   return new Promise((resolve, reject) => {
 36     executor.Execute('id', []).then(output => {
 37       if (output.stderr) {
 38         reject(`Failed to identify current user: ${output.stderr}`);
 39         return;
 40       }
 41 
 42       let outStr = output.stdout.trim();
 43       let parts = outStr.split(' ');
 44 
 45       // UID
 46       let uidParts = parts[0].split('=')[1];
 47       let uid = parseInt(uidParts.split('(')[0]);
 48       let username = uidParts.split('(')[1].replace(')', '');
 49 
 50       // GID
 51       let gidParts = parts[1].split('=')[1];
 52       let gid = parseInt(gidParts.split('(')[0]);
 53 
 54       // GROUPS
 55       let groupsParts = parts[2].split('=')[1].split(',');
 56 
 57       let groups = [];
 58       groupsParts.forEach(gStr => {
 59         let groupId = parseInt(gStr.split('(')[0]);
 60         let groupName = gStr.split('(')[1].slice(0, -1);
 61         groups.push({ gid: groupId, name: groupName });
 62       });
 63 
 64       let info = {
 65         username: username,
 66         uid: uid,
 67         gid: gid,
 68         groups: groups
 69       };
 70 
 71       resolve(info);
 72     }).catch(error => reject(`Failed to identify current user: ${error}`));
 73   });
 74 }

 76 /**
 77  * Get current user info.
 78  * @param {string} username
 79  * @param {Command} executor Command object that will execute the command.
 80  * @returns {Promise<{username: string, uid: number, gid: number, groups: Array<{gid: number, name: string}> >}>} Returns a promise. If it resolves, it returns an object. Else, it returns an error.
 81  */
 82 function OtherUser(username, executor) {
 83   let usernameError = VALIDATE.IsStringInput(username);
 84   if (usernameError)
 85     return Promise.reject(`Failed to identify other user: username is ${usernameError}`);
 86 
 87   if (!executor)
 88     return Promise.reject(`Failed to identify other user: Executor is required`);
 89 
 90   return new Promise((resolve, reject) => {
 91     executor.Execute('id', [username], executor).then(output => {
 92       if (output.stderr) {
 93         if (output.stderr.toLowerCase().includes('no such user'))
 94           reject(`Failed to identify other user: No such user: ${username}`);
 95         else
 96           reject(`Failed to identify other user: ${output.stderr}`);
 97         return;
 98       }
 99 
100       let outStr = output.stdout.trim();
101       let parts = outStr.split(' ');
102 
103       // UID
104       let uidParts = parts[0].split('=')[1];
105       let uid = parseInt(uidParts.split('(')[0]);
106 
107       // GID
108       let gidParts = parts[1].split('=')[1];
109       let gid = parseInt(gidParts.split('(')[0]);
110 
111       // GROUPS
112       let groupsParts = parts[2].split('=')[1].split(',');
113 
114       let groups = [];
115       groupsParts.forEach(gStr => {
116         let groupId = parseInt(gStr.split('(')[0]);
117         let groupName = gStr.split('(')[1].slice(0, -1);
118         groups.push({ gid: groupId, name: groupName });
119       });
120 
121       let info = {
122         username: username,
123         uid: uid,
124         gid: gid,
125         groups: groups
126       };
127       resolve(info);
128     }).catch(error => reject(`Failed to identify other user: ${error}`));
129   });
130 }
