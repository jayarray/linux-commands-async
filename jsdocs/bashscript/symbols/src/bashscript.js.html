1 let FILE = require('./file.js');
  2 let VALIDATE = require('./validate.js');
  3 
  4 //----------------------------------------
  5 // BASH SCRIPT
  6 
  7 /**
  8  * Create a basch script file.
  9  * @param {string} path Location where file should be created.
 10  * @param {string} content Text to be written in file. (Omit '#!/bin/bash' as it is first line in file).
 11  * @param {Command} executor Command object that will execute the command.
 12  * @returns {Promise} Returns a promise that will resolve if successful, otherwise it returns an error.
 13  */
 14 function Create(path, content, executor) {
 15   let pathError = VALIDATE.IsStringInput(path);
 16   if (pathError)
 17     return Promise.reject(`Failed to create bash script: path is ${pathError}`);
 18 
 19   let contentError = VALIDATE.IsStringInput(content);
 20   if (contentError == 'null' || contentError == 'undefined' || contentError == 'not a string')
 21     return Promise.reject(`Failed to create bash script: content is ${contentError}`);
 22 
 23   if (!executor)
 24     return Promise.reject(`Failed to create bashscript: Executor is required`);
 25 
 26   return new Promise((resolve, reject) => {
 27     FILE.Create(path, `#!/bin/bash\n${content}`, executor).then(success => {
 28       FILE.MakeExecutable(path, executor).then(success => {
 29         resolve();
 30       }).catch(error => reject(`Failed to create bashscript: ${error}`));
 31     }).catch(error => reject(`Failed to create bashscript: ${error}`));
 32   });
 33 }
 34 
 35 /**
 36  * Creates bash script, executes it, and cleans it up.
 37  * @param {string} path Location where file should be created.
 38  * @param {string} content Text to be written in file. (Omit '#!/bin/bash' as it is first line in file).
 39  * @param {Command} executor Command object that will execute the command.
 40  * @returns {Promise<string>} Returns a promise. If it resolves, it returns a string with the output. Else, it returns an error.
 41  */
 42 function Execute(path, content, executor) {
 43   let pathError = VALIDATE.IsStringInput(path);
 44   if (pathError)
 45     return Promise.reject(`Failed to execute bash script: path is ${pathError}`);
 46 
 47   let contentError = VALIDATE.IsStringInput(content);
 48   if (contentError == 'null' || contentError == 'undefined' || contentError == 'not a string')
 49     return Promise.reject(`Failed to execute bash script: content is ${contentError}`);
 50 
 51   if (!executor)
 52     return Promise.reject(`Failed to execute bashscript: Executor is required`);
 53 
 54   return new Promise((resolve, reject) => {
 55     Create(path, content, executor).then(success => {
 56       executor.Execute(path, []).then(output => {
 57         resolve(output.stdout);
 58         FILE.Remove(path, executor).then(success => {
 59         }).catch(error => reject(`Failed to execute bash script: ${error}`));
 60       }).catch(error => reject(`Failed to execute bash script: ${error}`));
 61     }).catch(error => reject(`Failed to execute bash script: ${error}`));
 62   });
 63 }
